<script>
// Create atmosphere particles
function createParticles() {
const particles = document.getElementById('particles');
for (let i = 0; i < 50; i++) {
const particle = document.createElement('div');
particle.className = 'particle';
particle.style.left = Math.random() * 100 + '%';
particle.style.top = Math.random() * 100 + '%';
particle.style.animation = `float ${5 + Math.random() * 10}s infinite ease-in-out`;
particle.style.animationDelay = Math.random() * 5 + 's';
particles.appendChild(particle);
}

// Add animation style
const style = document.createElement('style');
style.textContent = `
@keyframes float {
0%, 100% { transform: translate(0, 0); opacity: 0.3; }
50% { transform: translate(${Math.random() * 20 - 10}px, ${Math.random() * 20 - 10}px); opacity: 0.6; }
}
`;
document.head.appendChild(style);
}

// Create floating eyes
function createEyes() {
const eyes = document.getElementById('eyes');
for (let i = 0; i < 15; i++) {
const eye = document.createElement('div');
eye.className = 'eye';
eye.style.left = Math.random() * 100 + '%';
eye.style.top = Math.random() * 100 + '%';
eye.style.animation = `glow ${2 + Math.random() * 3}s infinite ease-in-out`;
eye.style.animationDelay = Math.random() * 3 + 's';
eyes.appendChild(eye);
}

// Add glow animation
const style = document.createElement('style');
style.textContent = `
@keyframes glow {
0%, 100% { opacity: 0; transform: scale(1); }
50% { opacity: 0.7; transform: scale(1.2); }
}
`;
document.head.appendChild(style);
}

// Flickering lights effect
function flickerLights() {
const flicker = document.getElementById('flicker');
let count = 0;
const interval = setInterval(() => {
flicker.style.opacity = Math.random() * 0.3;
count++;
if (count > 20) {
clearInterval(interval);
flicker.style.opacity = 0;
}
}, 100);
}

// Show eyes in the darkness
function showEyes() {
const eyes = document.querySelectorAll('.eye');
eyes.forEach(eye => {
eye.style.opacity = 0.8;
});

setTimeout(() => {
eyes.forEach(eye => {
eye.style.opacity = 0;
});
}, 2000);
}

// Blood effect
function showBlood() {
const blood = document.getElementById('blood');
blood.style.opacity = 0.7;

setTimeout(() => {
blood.style.opacity = 0;
}, 3000);
}

// Heart pulse effect
function pulseEffect() {
const pulse = document.getElementById('pulse');
pulse.style.backgroundColor = '#ff0000';
pulse.style.opacity = 0.3;

let scale = 1;
const interval = setInterval(() => {
pulse.style.transform = `scale(${scale})`;
pulse.style.opacity = 0.4 - (scale * 0.1);
scale += 0.1;
if (scale > 4) {
clearInterval(interval);
pulse.style.opacity = 0;
}
}, 100);
}

// The main jumpscare function
function triggerJumpscare() {
// Disable button to prevent multiple clicks
document.getElementById('surprise').disabled = true;

// Initial effects
flickerLights();
showEyes();
showBlood();
pulseEffect();

// Build up to the jumpscare
setTimeout(() => {
// Screen flash
document.getElementById('flicker').style.backgroundColor = '#ff0000';
document.getElementById('flicker').style.opacity = 0.7;

// Scream sound
try {
const audioContext = new (window.AudioContext || window.webkitAudioContext)();
const oscillator = audioContext.createOscillator();
const gainNode = audioContext.createGain();

oscillator.type = 'sawtooth';
oscillator.frequency.setValueAtTime(200, audioContext.currentTime);
oscillator.frequency.exponentialRampToValueAtTime(1000, audioContext.currentTime + 0.3);

gainNode.gain.setValueAtTime(0.8, audioContext.currentTime);
gainNode.gain.exponentialRampToValueAtTime(0.01, audioContext.currentTime + 0.5);

oscillator.connect(gainNode);
gainNode.connect(audioContext.destination);

oscillator.start();
oscillator.stop(audioContext.currentTime + 0.5);
} catch (e) {
console.log('Web Audio API not supported');
}

// Show jumpscare image
const jumpscare = document.getElementById('jumpscare');
const jumpscareImg = document.getElementById('jumpscare-img');

jumpscare.style.opacity = 1;
jumpscare.style.pointerEvents = 'all';
jumpscareImg.style.opacity = 1;
jumpscareImg.style.transform = 'scale(1.2)';

// Shake effect
document.body.style.animation = 'shake 0.5s';

}, 2000);

// Reset after jumpscare
setTimeout(() => {
const jumpscare = document.getElementById('jumpscare');
const jumpscareImg = document.getElementById('jumpscare-img');

jumpscare.style.opacity = 0;
jumpscare.style.pointerEvents = 'none';
jumpscareImg.style.opacity = 0;
jumpscareImg.style.transform = 'scale(0.9)';

document.getElementById('flicker').style.opacity = 0;
document.body.style.animation = '';

// Show birthday message
alert("Just kidding! Happy Birthday! 🎂");

// Re-enable button
document.getElementById('surprise').disabled = false;
}, 4000);
}

// Add shake animation
const shakeStyle = document.createElement('style');
shakeStyle.textContent = `
@keyframes shake {
0% { transform: translate(0, 0); }
20% { transform: translate(-10px, 0); }
40% { transform: translate(10px, 0); }
60% { transform: translate(-10px, 0); }
80% { transform: translate(10px, 0); }
100% { transform: translate(0, 0); }
}
`;
document.head.appendChild(shakeStyle);

// Initialize the page
window.onload = function() {
createParticles();
createEyes();

// Add event listener to the button
document.getElementById('surprise').addEventListener('click', triggerJumpscare);

// Occasional flicker for atmosphere
setInterval(() => {
if (Math.random() > 0.7) {
flickerLights();
}
}, 5000);
};
</script>
